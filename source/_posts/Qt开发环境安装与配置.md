```bash
# Install Qt-5.8.0
cd ~/Downloads
wget http://mirror.bit.edu.cn/qtproject/archive/qt/5.8/5.8.0/qt-opensource-linux-x64-5.8.0.run
sudo chmod +x qt-opensource-linux-x64-5.8.0.run 
./qt-opensource-linux-x64-5.8.0.run
## Install g++
sudo apt-get install build-essential
## Install generic front configration library runtime
sudo apt-get install libfontconfig1
```
